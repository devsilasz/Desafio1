package com.santander.agencia.config;

import io.swagger.v3.oas.models.OpenAPI;
import io.swagger.v3.oas.models.info.Contact;
import io.swagger.v3.oas.models.info.Info;
import io.swagger.v3.oas.models.info.License;
import io.swagger.v3.oas.models.servers.Server;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.List;

/**
 * Configuração do Swagger/OpenAPI para documentação da API.
 * 
 * Esta configuração define as informações da API, incluindo título, descrição,
 * versão e informações de contato, além de configurar os servidores disponíveis.
 */
@Configuration
public class SwaggerConfig {

    @Bean
    public OpenAPI customOpenAPI() {
        return new OpenAPI()
                .info(new Info()
                        .title("API de Agências - Santander")
                        .description("""
                            API REST para cadastro e consulta de distâncias de agências do Santander.
                            
                            ## Funcionalidades
                            - **Cadastro de Agências**: Cadastra novas agências com coordenadas X e Y
                            - **Consulta de Distâncias**: Retorna agências ordenadas por proximidade
                            - **Performance Otimizada**: Implementa otimizações para grandes volumes de dados
                            - **Cache Inteligente**: Melhora a performance das consultas frequentes
                            
                            ## Endpoints Principais
                            - `POST /desafio/cadastrar` - Cadastra uma nova agência
                            - `GET /desafio/distancia` - Busca agências próximas a uma posição
                            
                            ## Exemplos de Uso
                            
                            ### Cadastrar Agência
                            ```json
                            POST /desafio/cadastrar
                            {
                                "posX": 10,
                                "posY": -5,
                                "nome": "Agência Centro",
                                "codigo": "001"
                            }
                            ```
                            
                            ### Buscar Agências Próximas
                            ```
                            GET /desafio/distancia?posX=-10&posY=5
                            ```
                            
                            ## Tecnologias Utilizadas
                            - Spring Boot 3.2.0
                            - Spring Data JPA
                            - H2 Database (em memória)
                            - Swagger/OpenAPI 3
                            - Maven
                            """)
                        .version("1.0.0")
                        .contact(new Contact()
                                .name("Equipe Santander")
                                .email("suporte@santander.com.br")
                                .url("https://www.santander.com.br"))
                        .license(new License()
                                .name("Santander Internal")
                                .url("https://www.santander.com.br")))
                .servers(List.of(
                        new Server()
                                .url("http://localhost:8080")
                                .description("Servidor de Desenvolvimento"),
                        new Server()
                                .url("https://api.santander.com.br")
                                .description("Servidor de Produção")
                ));
    }
}
