package com.santander.agencia.dto;

import com.fasterxml.jackson.annotation.JsonProperty;
import io.swagger.v3.oas.annotations.media.Schema;
import jakarta.validation.constraints.NotNull;
import jakarta.validation.constraints.DecimalMax;
import jakarta.validation.constraints.DecimalMin;

/**
 * DTO para requisição de cadastro de agência.
 * Contém as coordenadas X e Y da agência a ser cadastrada.
 */
@Schema(description = "Dados para cadastro de uma nova agência")
public class CadastroAgenciaRequest {

    @JsonProperty("posX")
    @NotNull(message = "Posição X é obrigatória")
    @DecimalMin(value = "-180.0", message = "Posição X deve ser maior ou igual a -180")
    @DecimalMax(value = "180.0", message = "Posição X deve ser menor ou igual a 180")
    @Schema(description = "Coordenada X da agência", example = "10.0", required = true)
    private Double posX;

    @JsonProperty("posY")
    @NotNull(message = "Posição Y é obrigatória")
    @DecimalMin(value = "-90.0", message = "Posição Y deve ser maior ou igual a -90")
    @DecimalMax(value = "90.0", message = "Posição Y deve ser menor ou igual a 90")
    @Schema(description = "Coordenada Y da agência", example = "-5.0", required = true)
    private Double posY;

    @Schema(description = "Nome da agência (opcional)", example = "Agência Centro")
    private String nome;

    @Schema(description = "Código da agência (opcional)", example = "001")
    private String codigo;

    // Construtores
    public CadastroAgenciaRequest() {}

    public CadastroAgenciaRequest(Double posX, Double posY) {
        this.posX = posX;
        this.posY = posY;
    }

    public CadastroAgenciaRequest(Double posX, Double posY, String nome, String codigo) {
        this.posX = posX;
        this.posY = posY;
        this.nome = nome;
        this.codigo = codigo;
    }

    // Getters e Setters
    public Double getPosX() {
        return posX;
    }

    public void setPosX(Double posX) {
        this.posX = posX;
    }

    public Double getPosY() {
        return posY;
    }

    public void setPosY(Double posY) {
        this.posY = posY;
    }

    public String getNome() {
        return nome;
    }

    public void setNome(String nome) {
        this.nome = nome;
    }

    public String getCodigo() {
        return codigo;
    }

    public void setCodigo(String codigo) {
        this.codigo = codigo;
    }

    @Override
    public String toString() {
        return "CadastroAgenciaRequest{" +
                "posX=" + posX +
                ", posY=" + posY +
                ", nome='" + nome + '\'' +
                ", codigo='" + codigo + '\'' +
                '}';
    }
}
