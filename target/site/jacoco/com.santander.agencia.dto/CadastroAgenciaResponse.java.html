package com.santander.agencia.dto;

import com.fasterxml.jackson.annotation.JsonProperty;
import io.swagger.v3.oas.annotations.media.Schema;
import java.time.LocalDateTime;

/**
 * DTO para resposta de cadastro de agência.
 * Contém os dados da agência cadastrada com sucesso.
 */
@Schema(description = "Resposta do cadastro de agência")
public class CadastroAgenciaResponse {

    @JsonProperty("id")
    @Schema(description = "ID único da agência", example = "1")
    private Long id;

    @JsonProperty("nome")
    @Schema(description = "Nome da agência", example = "AGENCIA_1")
    private String nome;

    @JsonProperty("codigo")
    @Schema(description = "Código da agência", example = "001")
    private String codigo;

    @JsonProperty("posX")
    @Schema(description = "Coordenada X da agência", example = "10.0")
    private Double posX;

    @JsonProperty("posY")
    @Schema(description = "Coordenada Y da agência", example = "-5.0")
    private Double posY;

    @JsonProperty("dataCriacao")
    @Schema(description = "Data de criação da agência")
    private LocalDateTime dataCriacao;

    @JsonProperty("mensagem")
    @Schema(description = "Mensagem de sucesso", example = "Agência cadastrada com sucesso!")
    private String mensagem;

    // Construtores
    public CadastroAgenciaResponse() {}

    public CadastroAgenciaResponse(Long id, String nome, Double posX, Double posY, LocalDateTime dataCriacao) {
        this.id = id;
        this.nome = nome;
        this.posX = posX;
        this.posY = posY;
        this.dataCriacao = dataCriacao;
        this.mensagem = "Agência cadastrada com sucesso!";
    }

    // Getters e Setters
    public Long getId() {
        return id;
    }

    public void setId(Long id) {
        this.id = id;
    }

    public String getNome() {
        return nome;
    }

    public void setNome(String nome) {
        this.nome = nome;
    }

    public String getCodigo() {
        return codigo;
    }

    public void setCodigo(String codigo) {
        this.codigo = codigo;
    }

    public Double getPosX() {
        return posX;
    }

    public void setPosX(Double posX) {
        this.posX = posX;
    }

    public Double getPosY() {
        return posY;
    }

    public void setPosY(Double posY) {
        this.posY = posY;
    }

    public LocalDateTime getDataCriacao() {
        return dataCriacao;
    }

    public void setDataCriacao(LocalDateTime dataCriacao) {
        this.dataCriacao = dataCriacao;
    }

    public String getMensagem() {
        return mensagem;
    }

    public void setMensagem(String mensagem) {
        this.mensagem = mensagem;
    }

    @Override
    public String toString() {
        return "CadastroAgenciaResponse{" +
                "id=" + id +
                ", nome='" + nome + '\'' +
                ", codigo='" + codigo + '\'' +
                ", posX=" + posX +
                ", posY=" + posY +
                ", dataCriacao=" + dataCriacao +
                ", mensagem='" + mensagem + '\'' +
                '}';
    }
}
