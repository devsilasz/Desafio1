package com.santander.agencia.dto;

import com.fasterxml.jackson.annotation.JsonProperty;
import io.swagger.v3.oas.annotations.media.Schema;
import java.util.LinkedHashMap;
import java.util.Map;

/**
 * DTO para resposta de consulta de distâncias.
 * Contém um mapa ordenado das agências e suas respectivas distâncias.
 */
@Schema(description = "Resposta da consulta de distâncias das agências")
public class DistanciaResponse {

    @JsonProperty("posicaoUsuario")
    @Schema(description = "Posição do usuário que fez a consulta")
    private PosicaoUsuario posicaoUsuario;

    @JsonProperty("agencias")
    @Schema(description = "Mapa das agências ordenadas por distância (mais próxima primeiro)")
    private Map<String, String> agencias;

    @JsonProperty("totalAgencias")
    @Schema(description = "Total de agências encontradas", example = "3")
    private Integer totalAgencias;

    @JsonProperty("agenciaMaisProxima")
    @Schema(description = "Nome da agência mais próxima", example = "AGENCIA_2")
    private String agenciaMaisProxima;

    @JsonProperty("menorDistancia")
    @Schema(description = "Menor distância encontrada", example = "2.2")
    private Double menorDistancia;

    // Construtores
    public DistanciaResponse() {
        this.agencias = new LinkedHashMap<>();
    }

    public DistanciaResponse(PosicaoUsuario posicaoUsuario) {
        this();
        this.posicaoUsuario = posicaoUsuario;
    }

    // Métodos utilitários
    public void adicionarAgencia(String nomeAgencia, Double distancia) {
        String distanciaFormatada = String.format("distancia = %.2f", distancia).replace(",", ".");
        this.agencias.put(nomeAgencia, distanciaFormatada);
        
        if (this.agenciaMaisProxima == null || distancia < this.menorDistancia) {
            this.agenciaMaisProxima = nomeAgencia;
            this.menorDistancia = distancia;
        }
    }

    public void finalizar() {
        this.totalAgencias = this.agencias.size();
    }

    // Getters e Setters
    public PosicaoUsuario getPosicaoUsuario() {
        return posicaoUsuario;
    }

    public void setPosicaoUsuario(PosicaoUsuario posicaoUsuario) {
        this.posicaoUsuario = posicaoUsuario;
    }

    public Map<String, String> getAgencias() {
        return agencias;
    }

    public void setAgencias(Map<String, String> agencias) {
        this.agencias = agencias;
    }

    public Integer getTotalAgencias() {
        return totalAgencias;
    }

    public void setTotalAgencias(Integer totalAgencias) {
        this.totalAgencias = totalAgencias;
    }

    public String getAgenciaMaisProxima() {
        return agenciaMaisProxima;
    }

    public void setAgenciaMaisProxima(String agenciaMaisProxima) {
        this.agenciaMaisProxima = agenciaMaisProxima;
    }

    public Double getMenorDistancia() {
        return menorDistancia;
    }

    public void setMenorDistancia(Double menorDistancia) {
        this.menorDistancia = menorDistancia;
    }

    @Override
    public String toString() {
        return "DistanciaResponse{" +
                "posicaoUsuario=" + posicaoUsuario +
                ", totalAgencias=" + totalAgencias +
                ", agenciaMaisProxima='" + agenciaMaisProxima + '\'' +
                ", menorDistancia=" + menorDistancia +
                '}';
    }

    /**
     * Classe interna para representar a posição do usuário.
     */
    @Schema(description = "Posição do usuário")
    public static class PosicaoUsuario {
        
        @JsonProperty("posX")
        @Schema(description = "Coordenada X do usuário", example = "-10.0")
        private Double posX;

        @JsonProperty("posY")
        @Schema(description = "Coordenada Y do usuário", example = "5.0")
        private Double posY;

        // Construtores
        public PosicaoUsuario() {}

        public PosicaoUsuario(Double posX, Double posY) {
            this.posX = posX;
            this.posY = posY;
        }

        // Getters e Setters
        public Double getPosX() {
            return posX;
        }

        public void setPosX(Double posX) {
            this.posX = posX;
        }

        public Double getPosY() {
            return posY;
        }

        public void setPosY(Double posY) {
            this.posY = posY;
        }

        @Override
        public String toString() {
            return "PosicaoUsuario{" +
                    "posX=" + posX +
                    ", posY=" + posY +
                    '}';
        }
    }
}
